    <script>
        function handleResult(element, isCorrect) {
            // Remove previous classes in case user clicks multiple times
            element.classList.remove('clicked-correct', 'clicked-wrong');
            
            if (isCorrect) {
                element.classList.add('clicked-correct');
            } else {
                element.classList.add('clicked-wrong');
                // Auto-hide wrong answer after a second so they can try again
                setTimeout(() => {
                    element.classList.remove('clicked-wrong');
                }, 1000);
            }
        }
    </script>
